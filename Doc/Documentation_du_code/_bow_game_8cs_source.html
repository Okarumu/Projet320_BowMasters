193 private bool CollisionsTower(Ball ball, Tower tower)
194 {
195 // parcourt toutes les pièces de la tour
196 foreach (TowerPiece piece in tower.Pieces)
197 {
198 // balle au même endroit qu'une pièce + la pièce n'est pas déjà détruite
199 if (ball.ActualPosition.X == piece.Position.X && ball.ActualPosition.Y == piece.Position.Y && !piece.IsDestroyed)
200 {
201 // détruit la pièce
202 piece.DestroyPiece();
203 // indique qu'elle a été touchée
204 return true;
205 }
206 }
207 // pas de tour touchée
208 return false;
209 }

299 private void ThrowBall(Ball ball, Player thrower, Player ennemy, Tower myTower, Tower ennemyTower)
300 {
301 // bool de controle pour terminer le tour
302 bool endTurn = false;
303 // début du timer (référence à quand il a commencé)
304 DateTime startTime = DateTime.Now;
305 // le timer est égal à 0
306 double time = 0;
307
308 // boucle do...while qui affiche la balle en fonction du temps et l'arrete si elle touche un élément
309 do
310 {
311 // on update la position de la balle en fonction du temps
312 ball.UpdateBallPosition(time);
313 // on augmente le timer
314 time = (DateTime.Now - startTime).TotalSeconds;
315 // on affiche la balle
316 ball.DisplayBallInTime();
317
318 // on réaffiche les informations des personnages au cas où la balle les a effacés
319 ennemy.DisplayInfo();
320 thrower.DisplayInfo();
321
322 // s'il y a une collision avec l'ennemy
323 if (CollisionsPlayer(ball, ennemy))
324 {
325 // on fait le son de hit
326 SoundEffect.PlaySound("hitPlayer");
327 Thread.Sleep(100);
328
329 // on raffiche le joueur en lui enlevant de la vie
330 ennemy.DisplayStanding();
331 ennemy.TakeDamage(_AMOUNT_OF_DAMAGE);
332 // on rajoute du score au lanceur
333 thrower.Score += _SCORE_ADDER * 3;
334
335 // on raffiche les infos des joueurs updatés
336 ennemy.DisplayInfo();
337 thrower.DisplayInfo();
338
339 // on indique que le tour est fini
340 endTurn = true;
341 }
342 // s'il y a une collision avec sa propre tour
343 else if (CollisionsTower(ball, myTower))
344 {
345 // on fait le son de cassage de tour
346 SoundEffect.PlaySound("hitTower");
347 Thread.Sleep(100);
348 // on enlève du score au joueur et on le remontre
349 thrower.Score -= _SCORE_ADDER / 2;
350 thrower.DisplayInfo();
351
352 // on indique que le tour est fini
353 endTurn = true;
354 }
355 // s'il y a une collision avec la tour ennemie
356 else if (CollisionsTower(ball, ennemyTower))
357 {
358 // on fait le son de cassage de tour
359 SoundEffect.PlaySound("hitTower");
360 Thread.Sleep(100);
361
362 // on ajoute du score au lanceur et on l'update
363 thrower.Score += _SCORE_ADDER;
364 thrower.DisplayInfo();
365
366 // on indique que le tour est fini
367 endTurn = true;
368 }
369
370 Thread.Sleep(10);
371
372 // on efface la balle
373 ball.ErasePreviousBall();
374
375 } // tant que la balle est dans le jeu et que le tour n'est pas fini
376 while (CheckBallInGame(ball) && !endTurn);
377 }
