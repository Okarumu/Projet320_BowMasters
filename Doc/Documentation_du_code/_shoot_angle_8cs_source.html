188 public double UpdateBallAngle()
189 {
190 // on se met en blanc
191 Console.ForegroundColor = ConsoleColor.White;
192
193 // boucle jusqu'à ce que l'utilisateur appuie sur espace
194 while (true)
195 {
196 // comme on commence, ça veut dire que les points montent
197 _goingUp = true;
198 // de l'angle min à l'angle max
199 for (_angle = _minimum_angle; _angle < _maximum_angle; _angle += _ANGLE_DIFFERENCE)
200 {
201 // si l'utilisateur appuie sur espace
202 if ((GetAsyncKeyState(_VK_SPACE) & 0x8000) != 0)
203 {
204 // retourne l'angle
205 return _angle;
206 }
207 // si l'angle est dans les 5 angles possibles (0, 22.5, 45, 67.5 ou 90 pour tirer à droite)
208 if (_angle % _ANGLE_DIVIDER == 0)
209 {
210 // on efface le modèle
211 EraseModel();
212 // on affiche la nouvelle position
213 DisplayModel();
214 }
215 // petit temps d'attente
216 Thread.Sleep(10);
217 }
218 // on indique que les points descendent maintenant
219 _goingUp = false;
220 // de l'angle max à l'angle min
221 for (_angle = _maximum_angle; _angle > _minimum_angle; _angle -= _ANGLE_DIFFERENCE)
222 {
223 // si l'utilisatuer appuie sur espace
224 if ((GetAsyncKeyState(_VK_SPACE) & 0x8000) != 0)
225 {
226 // retourne l'angle
227 return _angle;
228 }
229 // si l'angle est dans les 5 angles possibles (0, 22.5, 45, 67.5 ou 90 pour tirer à droite)
