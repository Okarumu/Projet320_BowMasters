210 private bool CollisionsTower(Ball ball, Tower tower)
211 {
212 // parcourt toutes les pièces de la tour
213 foreach (TowerPiece piece in tower.Pieces)
214 {
215 // balle au même endroit qu'une pièce + la pièce n'est pas déjà détruite
216 if (ball.ActualPosition.X == piece.Position.X && ball.ActualPosition.Y == piece.Position.Y && !piece.IsDestroyed)
217 {
218 // détruit la pièce
219 piece.DestroyPiece();
220 // indique qu'elle a été touchée
221 return true;
222 }
223 }
224 // pas de tour touchée
225 return false;
226 }

316 private void ThrowBall(Ball ball, Player thrower, Player ennemy, Tower myTower, Tower ennemyTower)
317 {
318 // bool de controle pour terminer le tour
319 bool endTurn = false;
320 // début du timer (référence à quand il a commencé)
321 DateTime startTime = DateTime.Now;
322 // le timer est égal à 0
323 double time = 0;
324
325 // boucle do...while qui affiche la balle en fonction du temps et l'arrete si elle touche un élément
326 do
327 {
328 // on update la position de la balle en fonction du temps
329 ball.UpdateBallPosition(time);
330 // on augmente le timer
331 time = (DateTime.Now - startTime).TotalSeconds;
332 // on affiche la balle
333 ball.DisplayBallInTime();
334
335 // on réaffiche les informations des personnages au cas où la balle les a effacés
336 ennemy.DisplayInfo();
337 thrower.DisplayInfo();
338
339 // s'il y a une collision avec l'ennemy
340 if (CollisionsPlayer(ball, ennemy))
341 {
342 // on fait le son de hit
343 SoundEffect.PlaySound("hitPlayer");
344 Thread.Sleep(100);
345
346 // on raffiche le joueur en lui enlevant de la vie
347 ennemy.DisplayStanding();
348 ennemy.TakeDamage(_AMOUNT_OF_DAMAGE);
349 // on rajoute du score au lanceur
350 thrower.Score += _SCORE_ADDER * 3;
351
352 // on raffiche les infos des joueurs updatés
353 ennemy.DisplayInfo();
354 thrower.DisplayInfo();
355
356 // on indique que le tour est fini
357 endTurn = true;
358 }
359 // s'il y a une collision avec sa propre tour
360 else if (CollisionsTower(ball, myTower))
361 {
362 // on fait le son de cassage de tour
363 SoundEffect.PlaySound("hitTower");
364 Thread.Sleep(100);
365 // on enlève du score au joueur et on le remontre
366 thrower.Score -= _SCORE_ADDER / 2;
367 thrower.DisplayInfo();
368
369 // on indique que le tour est fini
370 endTurn = true;
371 }
372 // s'il y a une collision avec la tour ennemie
373 else if (CollisionsTower(ball, ennemyTower))
374 {
375 // on fait le son de cassage de tour
376 SoundEffect.PlaySound("hitTower");
377 Thread.Sleep(100);
378
379 // on ajoute du score au lanceur et on l'update
380 thrower.Score += _SCORE_ADDER;
381 thrower.DisplayInfo();
382
383 // on indique que le tour est fini
384 endTurn = true;
385 }
386
387 Thread.Sleep(10);
388
389 // on efface la balle
390 ball.ErasePreviousBall();
391
392 } // tant que la balle est dans le jeu et que le tour n'est pas fini
393 while (CheckBallInGame(ball) && !endTurn);
394 }

427 {

439 Console.Write("Appuyez sur échap pour fermer le jeu ou sur entrée pour recommencer une partie");
440
441 // boucle qui continue tant que les joueurs ne répondent pas à la question
442 do
443 {
444 // réponse des joueurs
445 keyInputUser = Console.ReadKey(true);
446
447 // input entrée = recommencer
448 if (keyInputUser.Key == ConsoleKey.Enter)
449 {
450 _playAgain = true;
451 }
452 // input echap = arrêter
453 else if (keyInputUser.Key == ConsoleKey.Escape)
454 {
455 _playAgain = false;
456 }
457 } while (keyInputUser.Key != ConsoleKey.Enter && keyInputUser.Key != ConsoleKey.Escape); // tant que les joueurs n'ont pas spécifié s'ils recommencent ou s'ils arrêtent
458 }
