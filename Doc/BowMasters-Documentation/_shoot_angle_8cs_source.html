195 public double UpdateBallAngle()
196 {
197 // on se met en blanc
198 Console.ForegroundColor = ConsoleColor.White;
199
200 // boucle jusqu'à ce que l'utilisateur appuie sur espace
201 while (true)
202 {
203 // comme on commence, ça veut dire que les points montent
204 _goingUp = true;
205 // de l'angle min à l'angle max
206 for (_angle = _minimum_angle; _angle < _maximum_angle; _angle += _ANGLE_DIFFERENCE)
207 {
208 // si l'utilisateur appuie sur espace
209 if ((GetAsyncKeyState(_VK_SPACE) & 0x8000) != 0)
210 {
211 // retourne l'angle
212 return _angle;
213 }
214 // si l'angle est dans les 5 angles possibles (0, 22.5, 45, 67.5 ou 90 pour tirer à droite)
215 if (_angle % _ANGLE_DIVIDER == 0)
216 {
217 // on efface le modèle
218 EraseModel();
219 // on affiche la nouvelle position
220 DisplayModel();
221 }
222 // petit temps d'attente
223 Thread.Sleep(10);
224 }
225
226 // on indique que les points descendent maintenant
227 _goingUp = false;
228 // de l'angle max à l'angle min
229 for (_angle = _maximum_angle; _angle > _minimum_angle; _angle -= _ANGLE_DIFFERENCE)

231 // si l'utilisatuer appuie sur espace

233 {
234 // retourne l'angle
